const API_URL = 'https://script.google.com/macros/s/AKfycbzgtD_V4KLdiuTmFMecvk8nOge0NTtJ3MaPsU08t9U0Mzid2Q5xD6saTjeTFHws-6je6w/exec';

// ============================================================
// ESTADO
// ============================================================
const MESES = ['Janeiro','Fevereiro','Março','Abril','Maio','Junho',
               'Julho','Agosto','Setembro','Outubro','Novembro','Dezembro'];

let currentDate = new Date();
let currentMonth = currentDate.getMonth();
let currentYear = currentDate.getFullYear();
let allData = null;
let chartInstance = null;

// ============================================================
// NAVEGAÇÃO
// ============================================================
function changeMonth(delta) {
  currentMonth += delta;
  if (currentMonth > 11) { currentMonth = 0; currentYear++; }
  if (currentMonth < 0)  { currentMonth = 11; currentYear--; }
  renderDashboard();
}

function updateMonthLabel() {
  document.getElementById('monthLabel').textContent =
    `${MESES[currentMonth]} ${currentYear}`;
}

// ============================================================
// FORMATAÇÃO
// ============================================================
function formatBRL(value) {
  return new Intl.NumberFormat('pt-BR', {
    style: 'currency', currency: 'BRL'
  }).format(value || 0);
}

function parseDate(val) {
  if (!val) return null;
  if (val instanceof Date) return val;
  const d = new Date(val);
  return isNaN(d) ? null : d;
}

// ============================================================
// FETCH
// ============================================================
async function fetchData() {
  document.getElementById('loading').style.display = 'block';
  document.getElementById('main').style.display = 'none';
  document.getElementById('statusDot').style.background = '#555';

  try {
    const res = await fetch(API_URL);
    const json = await res.json();
    if (!json.success) throw new Error(json.error);
    allData = json.data;
    document.getElementById('statusDot').style.background = 'var(--receita)';
    document.getElementById('statusDot').style.boxShadow = '0 0 8px var(--receita)';
  } catch (err) {
    document.getElementById('loading').innerHTML =
      `<div style="color:var(--despesa)">Erro ao carregar dados.<br><small>${err.message}</small></div>`;
    return;
  }

  document.getElementById('loading').style.display = 'none';
  document.getElementById('main').style.display = 'block';
  renderDashboard();
}

// ============================================================
// RENDER
// ============================================================
function renderDashboard() {
  updateMonthLabel();

  const { transacoes, categorias_menos, categorias_mais } = allData;

  // Mapas de ID → nome
  const mapMenos = {};
  categorias_menos.forEach(c => mapMenos[c.id] = c.nome);
  const mapMais = {};
  categorias_mais.forEach(c => mapMais[c.id] = c.nome);

  // Filtrar por mês/ano atual
  const doMes = transacoes.filter(t => {
    const d = parseDate(t.data);
    return d && d.getMonth() === currentMonth && d.getFullYear() === currentYear;
  });

  const despesas = doMes.filter(t => t.tipo === 'Menos graninha (-)');
  const receitas = doMes.filter(t => t.tipo === 'Mais graninha (+)');

  const totalDespesas = despesas.reduce((s, t) => s + (parseFloat(t.valor) || 0), 0);
  const totalReceitas = receitas.reduce((s, t) => s + (parseFloat(t.valor) || 0), 0);
  const saldo = totalReceitas - totalDespesas;

  // Agrupar despesas por categoria
  const porCategoria = {};
  despesas.forEach(t => {
    const nome = mapMenos[t.categoria_menos] || 'Sem categoria';
    porCategoria[nome] = (porCategoria[nome] || 0) + (parseFloat(t.valor) || 0);
  });

  const catSorted = Object.entries(porCategoria)
    .sort((a, b) => b[1] - a[1]);

  // Agrupar receitas por categoria
  const porCategoriaReceita = {};
  receitas.forEach(t => {
    const nome = mapMais[t.categoria_mais] || 'Sem categoria';
    porCategoriaReceita[nome] = (porCategoriaReceita[nome] || 0) + (parseFloat(t.valor) || 0);
  });

  const catReceitaSorted = Object.entries(porCategoriaReceita)
    .sort((a, b) => b[1] - a[1]);

  // Evolução mensal (últimos 6 meses)
  const evolucao = getEvolucao(transacoes, 6);

  // Renderizar HTML
  document.getElementById('main').innerHTML = `
    <!-- KPIs -->
    <div class="kpi-grid">
      <div class="kpi-card receita">
        <div class="kpi-label">Receitas</div>
        <div class="kpi-value">${formatBRL(totalReceitas)}</div>
      </div>
      <div class="kpi-card despesa">
        <div class="kpi-label">Despesas</div>
        <div class="kpi-value">${formatBRL(totalDespesas)}</div>
      </div>
      <div class="kpi-card saldo">
        <div>
          <div class="kpi-label">Saldo do mês</div>
          <div class="kpi-value" style="color:${saldo >= 0 ? 'var(--receita)' : 'var(--despesa)'}">
            ${formatBRL(saldo)}
          </div>
        </div>
        <div class="kpi-icon">${saldo >= 0 ? '📈' : '📉'}</div>
      </div>
    </div>

    <!-- Gastos por Categoria -->
    <div class="section">
      <div class="section-title">Gastos por categoria</div>
      ${catSorted.length === 0
        ? '<div class="empty">Nenhuma despesa este mês</div>'
        : `<div class="cat-list">${catSorted.map(([nome, val]) => `
          <div class="cat-item">
            <div class="cat-bar-wrap">
              <div class="cat-name">
                <span>${nome}</span>
                <span class="cat-amount">${formatBRL(val)}</span>
              </div>
              <div class="cat-bar-bg">
                <div class="cat-bar-fill" style="width:${(val/totalDespesas*100).toFixed(1)}%"></div>
              </div>
              <div class="cat-pct">${(val/totalDespesas*100).toFixed(1)}%</div>
            </div>
          </div>`).join('')}
        </div>`
      }
    </div>

    <!-- Receitas por Categoria -->
    <div class="section">
      <div class="section-title">Receitas por categoria</div>
      ${catReceitaSorted.length === 0
        ? '<div class="empty">Nenhuma receita este mês</div>'
        : `<div class="cat-list">${catReceitaSorted.map(([nome, val]) => `
          <div class="cat-item">
            <div class="cat-bar-wrap">
              <div class="cat-name">
                <span>${nome}</span>
                <span class="cat-amount" style="color:var(--receita)">${formatBRL(val)}</span>
              </div>
              <div class="cat-bar-bg">
                <div class="cat-bar-fill" style="width:${(val/totalReceitas*100).toFixed(1)}%;background:var(--receita)"></div>
              </div>
              <div class="cat-pct">${(val/totalReceitas*100).toFixed(1)}%</div>
            </div>
          </div>`).join('')}
        </div>`
      }
    </div>

    <!-- Evolução mensal -->
    <div class="section">
      <div class="section-title">Evolução — últimos 6 meses</div>
      <div class="chart-wrap">
        <canvas id="evolChart"></canvas>
      </div>
    </div>

    <footer>Atualizado agora • Graninha 💜</footer>
  `;

  // Renderizar gráfico
  renderChart(evolucao);
}

// ============================================================
// EVOLUÇÃO MENSAL
// ============================================================
function getEvolucao(transacoes, meses) {
  const result = [];
  for (let i = meses - 1; i >= 0; i--) {
    const d = new Date(currentYear, currentMonth - i, 1);
    const m = d.getMonth();
    const y = d.getFullYear();
    const label = `${MESES[m].slice(0,3)}/${String(y).slice(2)}`;

    const filtrado = transacoes.filter(t => {
      const td = parseDate(t.data);
      return td && td.getMonth() === m && td.getFullYear() === y;
    });

    const despesa = filtrado
      .filter(t => t.tipo === 'Menos graninha (-)')
      .reduce((s, t) => s + (parseFloat(t.valor) || 0), 0);
    const receita = filtrado
      .filter(t => t.tipo === 'Mais graninha (+)')
      .reduce((s, t) => s + (parseFloat(t.valor) || 0), 0);

    result.push({ label, despesa, receita });
  }
  return result;
}

function renderChart(evolucao) {
  if (chartInstance) chartInstance.destroy();
  const ctx = document.getElementById('evolChart').getContext('2d');
  chartInstance = new Chart(ctx, {
    type: 'bar',
    data: {
      labels: evolucao.map(e => e.label),
      datasets: [
        {
          label: 'Receitas',
          data: evolucao.map(e => e.receita),
          backgroundColor: 'rgba(74,222,128,0.7)',
          borderRadius: 6,
        },
        {
          label: 'Despesas',
          data: evolucao.map(e => e.despesa),
          backgroundColor: 'rgba(248,113,113,0.7)',
          borderRadius: 6,
        }
      ]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: {
          labels: { color: '#7a7a96', font: { size: 11 }, boxWidth: 12 }
        },
        tooltip: {
          callbacks: {
            label: ctx => formatBRL(ctx.raw)
          }
        }
      },
      scales: {
        x: { ticks: { color: '#7a7a96', font: { size: 10 } }, grid: { display: false } },
        y: {
          ticks: {
            color: '#7a7a96', font: { size: 10 },
            callback: v => 'R$' + v.toLocaleString('pt-BR')
          },
          grid: { color: 'rgba(255,255,255,0.05)' }
        }
      }
    }
  });
}
